Նույնիսկ ցածր չափանիշներին աղետների ֆիլմերի, ներեցեք ինձ թույլ տվեք ուղղել ինձ, նույնիսկ ցածր չափանիշներին կազմել է հեռուստատեսային աղետների ֆիլմերի Սա իսկապես սարսափելի. Որտեղ եմ սկսել: <br /> <br /> The պատնեշ մոդելավորող կարող է եւս մեկ անգամ տեսել է մի լուսանկար է պատվարի, բայց ես կասկածում եմ: Շատ Ամբարտակների, հատկապես խոշոր նրանք, որ էլեկտրաէներգիա պետք ախ, ես չգիտեմ, թե մի ատոմակայանի մոտակայքում, որոշ ջրարգելակ դարպասների համար ջրի առաջադրվել միջոցով, heck նույնիսկ բարձր լարվածությունը էլեկտրական հեռագիր կամ երկու. <br /> <br /> The պատնեշ է նաեւ որոշ չափով պակաս: Երկու, հաշվել նրանց, երկու աշխատակիցներ աշխատակազմը ամբողջ պատնեշ, բոլոր երեք տեղաշարժերը: Եւ աշխատակիցները, ըստ երեւույթին, ներմուծվել է clichéd բռնված համաշխարհային պատերազմը երկու ֆիլմում, քանի որ նրանք հերոսաբար եւ բավականին անտեղի վաղուց գովաբանել մահվան տեսարաններ ամբողջական լուսանկարներով grandchildrenn լողացող կողմից: Heck նրանցից մեկը կարողանում է ստանալ նկարահանվել է պատնեշ հենց մի ճանապարհ, որ այլեւ նկարագրությունը: <br /> <br /> Հատուկ էֆեկտներ բաղկացած է հոսող ջուր բարդվել լուսանկարը ի քաղաքի մի եղանակով, որ ստիպում է sixties Godzilla ֆիլմի տեսք է Matrix. Երեք տարեկան նկարչություն կապույտ նկար վրա ֆիլմի ֆոնդի կլիներ տվել ավելի լավ արդյունքներ: Քանի որ աղետի դրամական կրակոցներ են անարժեք, թե ինչպես է մնացածը, որ ֆիլմի. <br /> <br /> Ավելորդ է ասել, որ սցենարը եւ գործող հետեւում են հատուկ էֆեկտներ հանգեցնել: Սա ոչ ադամանդի է կոշտ.
